module MyPenultimate (penultimate) where

-- | Get second to last element of a list.
penultimate :: [a] -> Maybe a
penultimate []     = Nothing
penultimate [_]    = Nothing
penultimate [a,_]  = Just a
penultimate (_:as) = penultimate as
