module Lower (mkLower) where

import           Data.Char (isAlpha, isLower)

-- | Make a lowercase letter.
mkLower :: Char -> Either String Char
mkLower x
  | isLower x && isAlpha x = Right x
  | otherwise              = Left "Not lowercase"
